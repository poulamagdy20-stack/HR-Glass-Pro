rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function signedIn() { return request.auth != null; }
    function userDoc() { return get(/databases/$(database)/documents/users/$(request.auth.uid)); }
    function hasUser() { return signedIn() && exists(/databases/$(database)/documents/users/$(request.auth.uid)); }
    function role() { return hasUser() ? userDoc().data.role : null; }
    function company() { return hasUser() ? userDoc().data.companyId : null; }
    function isAdmin() { return role() in ['admin','hr','accountant']; }
    function isEmployee() { return role() == 'employee'; }
    function sameCompanyData() { return signedIn() && request.resource.data.companyId == company(); }
    function sameCompanyExisting() { return signedIn() && resource.data.companyId == company(); }

    match /users/{uid} {
      allow read: if signedIn() && (request.auth.uid == uid || isAdmin());
      allow create: if signedIn() && (request.auth.uid == uid || isAdmin());
      allow update: if signedIn() && (request.auth.uid == uid || isAdmin());
      allow delete: if isAdmin();
    }

    match /employees/{id} {
      allow read: if signedIn() && (isAdmin() || resource.data.uid == request.auth.uid);
      allow create: if isAdmin() && sameCompanyData();
      allow update, delete: if isAdmin() && sameCompanyExisting();
    }

    match /shifts/{id} {
      allow read: if signedIn() && resource.data.companyId == company();
      allow create: if isAdmin() && sameCompanyData();
      allow update, delete: if isAdmin() && sameCompanyExisting();
    }

    match /settings/{id} {
      allow read: if signedIn();
      allow write: if isAdmin();
    }

    match /attendance/{id} {
      allow read: if signedIn() && (isAdmin() || resource.data.uid == request.auth.uid);
      allow create: if signedIn() && request.resource.data.companyId == company() && (isAdmin() || request.resource.data.uid == request.auth.uid);
      allow update, delete: if isAdmin() && sameCompanyExisting();
    }

    match /requests/{id} {
      allow read: if signedIn() && (isAdmin() || resource.data.uid == request.auth.uid);
      allow create: if signedIn() && request.resource.data.companyId == company() && request.resource.data.uid == request.auth.uid;
      allow update, delete: if isAdmin() && sameCompanyExisting();
    }

    match /transactions/{id} {
      allow read: if signedIn() && (isAdmin() || resource.data.uid == request.auth.uid);
      allow create: if isAdmin() && request.resource.data.companyId == company();
      allow update: if isAdmin() && resource.data.companyId == company() && request.resource.data.companyId == company();
      allow delete: if isAdmin() && resource.data.companyId == company();
    }

    match /payroll/{id} {
      allow read: if signedIn() && (isAdmin() || resource.data.uid == request.auth.uid);
      allow write: if isAdmin();
    }
  }
}
